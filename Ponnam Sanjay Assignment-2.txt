>> select ename, job,comm  from emp,dept

4. How many employees work in Chicago?

>> select count(*) as 'employes in loaction',loc from emp,dept 
   where emp.deptno=dept.deptno and dept.loc='chicago'

19. write a subquery using case statement to display non sales if department is not sales 
    If salary of an employee is more than 1000, then display salary otherwise display 1000

>> select ename,
   (case
    when sal = any (select sal
    from emp
    where sal>1000) then sal  
    when sal = any (select sal
    from emp
    where sal<1000) then '1000' end) as 'status-result'
    from emp
    where  emp.job != 'salesman';

20. Display salary as low, medium high based on salary range Less than 1000 less 1000 to 2500 medium More than 2500 high

>> select ename, sal,
   (case
    when sal = any (select sal
    from emp
    where 1000>sal) then ' low sal'
    when sal = any (select sal
    from emp
    where sal between 1000 and 2500) then 'medium sal'
    when sal = any (select sal
    from emp
    where 2500<sal) then 'high sal' end) as 'range of sal'
    from emp;
